using FluentAssertions;
using Xunit;
using Shouldly;

namespace FluentAssertionsToShouldlyAnalyzer.Test.TestCases
{
    public class PersonTests
    {
        [Fact]
        public void IsAdult_Should_BeFalse_When_Age_Is_17()
